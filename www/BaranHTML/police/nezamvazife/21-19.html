ﭘﻴﻮﺳﺖ

ﺩﺭ ﺭﺍﺳﺘﺎﯼ ﺍﺟﺮﺍﯼ ﻣﺎﺩﻩ ۲۳ ﻓﺼﻞ ﺩﻭﻡ ﺁﻳﻴﻦ ﻧﺎﻣﻪ ﻣﻌﺎﻳﻨﻪ ﻭ ﻣﻌﺎﻓﻴﺖ ﻣﺸﻤﻮﻟﻴﻦ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﭘﺪﺭ ﻣﺸﻤﻮﻝ ﺩﺍﺭﺍﯼ ﺑﻴﻤﺎﺭﯼ ﻳﺎ ﻧﻘﺺ ﻋﻀﻮﯼ ﺑﻪ ﺷﺮﺡ ﺯﻳﺮ ﺑﺎﺷﺪ ﺍﺯ ﻧﻈﺮ ﺷﻮﺭﺍﯼ ﭘﺰﺷﮑﯽ ﻧﻴﺎﺯ ﺑﻪ ﻣﺮﺍﻗﺒﺖ ﻭ ﻧﮕﻬﺪﺍﺭﯼ ﺩﺍﺭﺩ:

۱- ﮐﻠﻴﻪ ﺑﺪﺧﻴﻤﯽ ﻫﺎﯼ ﺩﺳﺘﮕﺎﻩ ﻫﺎﯼ ﺑﺪﻥ ﮐﻪ ﻣﻮﺟﺐ ﻧﺎﺗﻮﺍﻧﯽ ﻭ ﺍﺯ ﮐﺎﺭﺍﻓﺘﺎﺩﮔﯽ ﻓﺮﺩ ﮔﺮﺩﺩ ﻭ ﻳﺎ ﭘﻴﺸﺮﻓﺘﻪ ﻭ ﻣﻘﺎﻭﻡ ﺑﻪ ﺩﺭﻣﺎﻥ ﺑﺎﺷﺪ ﻭ ﺑﻪ ﺗﺄﻳﻴﺪ ﻣﺮﺍﺟﻊ ﺫﯼ ﺻﻼﺡ ﺭﺳﻴﺪﻩ ﺑﺎﺷﺪ.

۲- ﺗﻮﻣﻮﺭﻫﺎ ﻭ ﺁﺩﻧﻮﭘﺎﺗﯽ ﻫﺎﯼ ﻣﺰﻣﻦ ﻭ ﻏﻴﺮ ﻗﺎﺑﻞ ﺩﺭﻣﺎﻥ ﻣﺪﻳﺎﺳﺘﻦ ﻫﻤﺮﺍﻩ ﺑﺎ ﻋﻼﺋﻢ ﻓﺸﺎﺭﯼ ﺑﺮﻣﺪﻳﺎﺳﺘﻦ.

ﺗﺒﺼﺮﻩ: ﺩﺭ ﻣﻮﺭﺩ ﺳﺮﻃﺎﻥ ﻫﺎﯼ ﻗﺎﺑﻞ ﺩﺭﻣﺎﻥ ﻭ ﻣﺤﺪﻭﺩ ﮐﻪ ﻣﺘﺎﺳﺘﺎﺯ ﻧﺪﺍﺩﻩ ﻭ ﻋﻮﺍﺭﺽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺍﻳﻦ ﻗﺴﻤﺖ ﻣﺼﺪﺍﻕ ﻧﺪﺍﺭﺩ.

۳- ﺟﺬﺍﻡ ﺑﺎ ﺗﺸﺨﻴﺺ ﻣﺘﺨﺼﺺ ﭘﻮﺳﺖ ﻭ ﺗﺄﻳﻴﺪ ﻣﺮﺍﮐﺰ ﻣﺒﺎﺭﺯﻩ ﺑﺎ ﺟﺬﺍﻡ ﺩﺭ ﻣﺮﺣﻠﻪ ﺍﻳﺰﻭﻟﻪ ﺷﺪﻥ ﺑﻴﻤﺎﺭ ﻭ ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺧﻄﺮ ﻭﺍﮔﻴﺮ ﺑﻴﻤﺎﺭﯼ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﻧﺤﻮ ﺩﺭ ﺍﻧﻮﺍﻋﯽ ﺍﺯ ﺑﻴﻤﺎﺭﯼ ﮐﻪ ﭘﻴﺸﺮﻓﺘﻪ ﺑﻮﺩﻩ ﻭ ﻋﻮﺍﺭﺽ ﺷﺪﻳﺪ ﭘﻮﺳﺘﯽ ﻳﺎ ﻋﺼﺒﯽ ﺩﺭ ﺷﺨﺺ ﻣﺒﺘﻼ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺑﺎﺷﺪ.

۴- ﺳﻴﻔﻠﻴﺲ ﻫﺎﯼ ﻋﺼﺒﯽ ﻳﺎ ﻗﻠﺒﯽ ﻋﺮﻭﻗﯽ ﺷﺪﻳﺪ ﻭ ﻳﺎ ﺳﻴﻔﻠﻴﺲ ﻫﺎﯼ ﺍﺳﺘﺨﻮﺍﻧﯽ ﮐﻪ ﺍﻳﺠﺎﺩ ﺿﺎﻳﻌﺎﺕ ﺍﻧﻬﺪﺍﻣﯽ ﺩﺭ ﺍﺳﺘﺨﻮﺍﻥ ﻳﺎ ﻣﻔﺎﺻﻞ ﮐﺮﺩﻩ ﺑﺎﺷﺪ.

۵- ﻣﺒﺘﻼﻳﺎﻥ ﺑﻪ ﺳﻞ ﻓﻌﺎﻝ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻣﺤﺪﻭﺩﻳﺖ ﻭ ﺍﺯ ﮐﺎﺭﺍﻓﺘﺎﺩﮔﯽ ﻓﺮﺩ ﮔﺮﺩﺩ ﺑﺎ ﺗﺄﻳﻴﺪ ﻣﺮﺍﮐﺰ ﺗﺨﺼﺼﯽ ﻣﺮﺑﻮﻃﻪ.

۶- ﺁﺭﺗﺮﻳﺖ ﻫﺎﯼ ﻣﺰﻣﻦ ﺗﻮﺃﻡ ﺑﺎ ﺗﻐﻴﻴﺮ ﺷﮑﻞ ﺩﺭ ﺍﺳﺘﺨﻮﺍﻥ ﻫﺎ ﻭ ﺟﻤﻮﺩ ﺩﺭ ﻣﻔﺎﺻﻞ ﻭ ﺳﺘﻮﻥ ﻓﻘﺮﺍﺕ ﺑﻪ ﺻﻮﺭﺕ ﭘﻴﺸﺮﻓﺘﻪ ﮐﻪ ﻣﺎﻧﻊ ﺍﻧﺠﺎﻡ ﻓﻌﺎﻟﻴﺖ ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺷﺪﻩ ﺑﺎﺷﺪ.

۷- ﺍﻧﻮﺍﻉ ﻧﻮﺭﻭﭘﺎﺗﯽ ﻫﺎ ﻭ ﻣﻴﻮﭘﺎﺗﯽ ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﻭ ﺛﺎﻧﻮﯼ ﺩﺭ ﻣﺮﺍﺣﻞ ﭘﻴﺸﺮﻓﺘﻪ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻧﺎﺗﻮﺍﻧﯽ ﻭ ﺍﺧﺘﻼﻝ ﺩﺭ ﺍﻳﺴﺘﺎﺩﻥ ﻭ ﺣﺮﮐﺖ ﺷﺪﻩ ﺑﺎﺷﺪ.

۸- ﻓﻠﺞ ﺍﻧﺪﺍﻡ ﻓﻮﻗﺎﻧﯽ ﻳﺎ ﺗﺤﺘﺎﻧﯽ ﺑﺪﻥ (ﻫﻤﯽ ﭘﻠﮋﯼ, ﭘﺎﺭﺍﻳﻠﮋﯼ) ﮐﻪ ﻣﻮﺟﺐ ﻧﺎﺗﻮﺍﻧﯽ ﺩﺭ ﺍﻋﻤﺎﻝ ﺭﻭﺯﺍﻧﻪ ﺷﺨﺺ ﺷﺪﻩ ﺑﺎﺷﺪ.

۹- ﭘﺎﺭﮐﻴﻨﺴﻮﻥ ﺩﺭ ﻣﺮﺍﺣﻞ ﭘﻴﺸﺮﻓﺘﻪ ﺑﺎ ﻋﻼﺋﻢ ﺑﺎﻟﻴﻨﯽ ﻭﺍﺿﺢ ﻭ ﺑﺎ ﺗﺄﻳﻴﺪ ﻣﺘﺨﺼﺺ ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺍﻋﺼﺎﺏ.

۱۰- ﻣﻮﻟﺘﻴﭙﻞ ﺍﺳﮑﻠﺮﻭﺯ ﺩﺭ ﻣﺮﺣﻠﻪ ﺍﺳﺘﻘﺮﺍﺭ ﮐﺎﻣﻞ.

۱۱- ﺍﻧﻮﺍﻉ ﺻﺮﻉ ﻣﻘﺎﻭﻡ ﺑﻪ ﺩﺭﻣﺎﻥ.

۱۲- ﮐﻠﻴﻪ ﯼ ﺿﺎﻳﻌﺎﺕ ﻋﺮﻭﻗﯽ ﻣﻐﺰﯼ ﮐﻪ ﻣﻮﺟﺐ ﻧﺎﺗﻮﺍﻧﯽ ﻓﺮﺩ ﮔﺮﺩﺩ.

۱۳- ﮐﻠﻴﻪ ﯼ ﺑﻴﻤﺎﺭﯼ ﻫﺎ, ﺗﺮﻭﻣﺎﻫﺎ ﻭ ﺍﻋﻤﺎﻝ ﺟﺮﺍﺣﯽ ﺭﻭﯼ ﺳﺘﻮﻥ ﻓﻘﺮﺍﺕ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﺍﺧﺘﻼﻻﺕ ﺍﺳﻔﻨﮕﺘﺮﯼ ﻭ ﻓﻘﺪﺍﻥ ﺍﺭﺯﺵ ﮐﺎﺭﯼ ﻳﮏ ﺍﻧﺪﺍﻡ ﺷﺪﻩ ﺑﺎﺷﺪ.

۱۴- ﻫﺮ ﮔﻮﻧﻪ ﺍﻋﻤﺎﻝ ﺟﺮﺍﺣﯽ ﺭﻭﯼ ﻣﻐﺰ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻋﻮﺍﺭﺽ ﻋﺼﺒﯽ ﻳﺎ ﺭﻭﺍﻧﯽ ﺷﺪﻳﺪ ﺑﻌﺪ ﺍﺯ ﻋﻤﻞ ﺷﺪﻩ ﺑﺎﺷﺪ.

۱۵- ﭘﺴﻴﮑﻮﺯﻫﺎ ﻭ ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺭﻭﺍﻧﯽ ﻣﺰﻣﻦ ﺑﺎ ﺳﺎﺑﻘﻪ ﻣﻌﺘﺒﺮ ﺑﺴﺘﺮﯼ ﻫﺎﯼ ﻣﺘﻌﺪﺩ ﺩﺭ ﺑﻴﻤﺎﺭﺳﺘﺎﻥ ﻫﺎﯼ ﺭﻭﺍﻧﯽ ﻭ ﻳﺎ ﺗﺄﻳﻴﺪ ﻣﺘﺨﺼﺺ ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺭﻭﺍﻧﯽ.

۱۶- ﻧﻘﺺ ﻋﻘﻼﻧﯽ ﻭ ﮐﻨﺪﺫﻫﻨﯽ ﺑﺎ ﮐﺴﺐ ﻧﻈﺮ ﺍﺯ ﻣﺘﺨﺼﺺ ﻣﺮﺑﻮﻃﻪ ﻭ ﻣﺮﺍﮐﺰ ﺗﻮﺍﻧﺒﺨﺸﯽ ﮐﻪ ﺿﺮﻳﺐ ﻫﻮﺷﯽ ﺑﻪ ۳۰% ﻭ ﮐﻤﺘﺮ ﻧﻘﺼﺎﻥ ﻳﺎﻓﺘﻪ ﺑﺎﺷﺪ (ﻣﻴﺰﺍﻥ ﮐﺎﺭﺍﻳﯽ ﺷﺨﺺ ﺍﺯ ﻧﻈﺮ ﻣﺮﺍﮐﺰ ﺗﻮﺍﻧﺒﺨﺸﯽ ۳۰% ﺑﺎﺷﺪ)

۱۷- ﺩﺭ ﻣﻮﺭﺩ ﻗﻄﻊ ﻳﺎ ﺍﺯ ﮐﺎﺭﺍﻓﺘﺎﺩﮔﯽ ﺍﻧﺪﺍﻡ ﻓﻮﻗﺎﻧﯽ ﻳﺎ ﺗﺤﺘﺎﻧﯽ ﺑﻪ ﺷﺮﺡ ﺯﻳﺮ ﺭﻓﺘﺎﺭ ﺷﻮﺩ:

ﺍﻟﻒ) ﻗﻄﻊ ﻳﮏ ﭘﺎ ﺍﺯ ﺯﺍﻧﻮ ﺑﻪ ﺑﺎﻻ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﻋﻤﻼ ﻳﮏ ﭘﺎ ﮐﺎﺭﺍﻳﯽ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻩ ﺑﺎﺷﺪ.
ﺏ) ﻗﻄﻊ ﻫﺮ ﺩﻭ ﭘﺎ ﺍﺯ ﻣﻔﺼﻞ ﻣﺘﺎﺗﺎﺭ ﺳﻮﻓﻼﻧﺠﻴﺎﻝ (MP) ﺑﻪ ﺑﺎﻻ.
ﺝ) ﻗﻄﻊ ﻳﮏ ﺩﺳﺖ ﺍﺯ ﺁﺭﻧﺞ ﺑﻪ ﺑﺎﻻ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﻳﮏ ﺩﺳﺖ ﻋﻤﻼ ﮐﺎﺭﺍﻳﯽ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻩ ﺑﺎﺷﺪ.
ﺩ) ﻗﻄﻊ ﻫﺮ ﺩﻭ ﺩﺳﺖ ﺍﺯ ﻫﺮ ﺟﺎ ﮐﻪ ﺑﺎﺷﺪ.

۱۸- ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﻣﺰﻣﻦ ﻭ ﺍﻧﺴﺪﺍﺩ ﺷﺪﻳﺪ ﺭﻳﻪ ﻫﻤﺮﺍﻩ ﮐﺎﻫﺶ ﻇﺮﻓﻴﺖ ﺗﻨﻔﺴﯽ ﺭﻳﻪ ﺑﻪ ﻣﻴﺰﺍﻥ ۵۰% ﻭ ﺑﻴﺸﺘﺮ.

۱۹- ﺗﻐﻴﻴﺮ ﺷﮑﻞ ﻫﺎﯼ ﺷﺪﻳﺪ ﻣﺎﺩﺭﺯﺍﺩﯼ ﺟﺪﺍﺭ ﻗﻔﺴﻪ ﺻﺪﺭﯼ ﮐﻪ ﻣﻮﺟﺐ ﮐﺎﻫﺶ ﻇﺮﻓﻴﺖ ﺗﻨﻔﺴﯽ ﺑﻪ ﻣﻴﺰﺍﻥ ۱۵% ﻭ ﻳﺎ ﮐﻤﺘﺮ ﻳﺎ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺐ ﺷﺪﻩ ﺑﺎﺷﺪ.

۲۰- ﺳﺎﺑﻘﻪ ﺍﻋﻤﺎﻝ ﺟﺮﺍﺣﯽ ﺭﻳﻪ ﻭ ﻗﻔﺴﻪ ﺻﺪﺭﯼ (ﺗﻮﺭﺍﮐﻮﭘﻼﺳﺘﯽ, ﺭﺯﮐﺴﻴﻮﻥ ﺭﻳﻪ) ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﮐﺎﻫﺶ ﻇﺮﻓﻴﺖ ﺗﻨﻔﺴﯽ ﺑﻪ ﻣﻴﺰﺍﻥ ۵۰% ﻭ ﻳﺎ ﮐﻤﺘﺮ ﺷﺪﻩ ﺑﺎﺷﺪ.

۲۱- ﻫﻴﭙﺮﺗﻴﺮﻭﺋﻴﺪﯼ ﺷﺪﻳﺪ ﻫﻤﺮﺍﻩ ﺑﺎ ﻋﻮﺍﺭﺽ ﻗﻠﺒﯽ, ﻋﺼﺒﯽ, ﭼﺸﻤﯽ ﻭ ﻻﻏﺮﯼ ﻣﻔﺮﻁ ﻭ ﻫﻴﭙﺮﺗﻴﺮﻭﺋﻴﺪﯼ ﭘﻴﺸﺮﻓﺘﻪ ﺑﺎ ﺍﻳﺠﺎﺩ ﻋﻮﺍﺭﺽ.

۲۲- ﺩﻳﺎﺑﺖ ﺷﻴﺮﻳﻦ ﻣﺰﻣﻦ ﻣﻘﺎﻭﻡ ﺑﻪ ﺩﺭﻣﺎﻥ ﺑﺎ ﻋﻮﺍﺭﺽ ﭼﺸﻤﯽ, ﻋﺼﺒﯽ ﻭ ﮐﻠﻴﻮﯼ ﭘﻴﺸﺮﻓﺘﻪ.

۲۳- ﺁﮐﺮﻭﻣﮕﺎﻟﯽ, ﮊﻳﮕﺎﻧﺘﻴﺴﻢ, ﮐﻮﺷﻴﻨﮓ, ﺁﺩﻳﺴﻮﻥ ﺩﺭ ﻫﺮ ﻣﻮﺭﺩ ﺑﻪ ﺷﺮﻃﯽ ﮐﻪ ﭘﻴﺸﺮﻓﺘﻪ ﺑﻮﺩﻩ ﻭ ﻋﻮﺍﺭﺽ ﺟﺴﻤﯽ ﻳﺎ ﺭﻭﺍﻧﯽ ﻳﺎ ﻋﻼﺋﻢ ﺑﺎﻟﻴﻨﯽ ﻭﺍﺿﺢ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺑﺎﺷﺪ ﺑﻪ ﻫﻤﺮﺍﻩ ﺳﻮﺍﺑﻖ ﻣﻌﺘﺒﺮ ﭘﺰﺷﮑﯽ ﻭ ﺑﺎ ﺗﺎﻳﻴﺪ ﻣﺘﺨﺼﺺ ﻏﺪﺩ ﻣﺘﺮﺷﺤﻪ ﺩﺍﺧﻠﯽ.

۲۴- ﭼﺎﻗﯽ ﻣﻔﺮﻁ ﺑﻪ ﻗﺴﻤﯽ ﮐﻪ ﻭﺯﻥ ﺷﺨﺺ ۸۰% ﻭ ﻳﺎ ﺑﺎﻻﺗﺮ ﺍﺯ ﻭﺯﻥ ﻣﻄﻠﻮﺏ =(۲+۱۰۰- ﻃﻮﻝ ﻗﺪ) ﻭ ﻳﺎ BMI > ۴۰ ﺑﺎﺷﺪ.

۲۵- ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﻣﺰﻣﻦ ﮐﺒﺪﯼ ﻫﻤﺮﺍﻩ ﺑﺎ ﻋﻮﺍﺭﺽ ﻣﺮﺑﻮﻃﻪ (ﺁﺳﻴﺖ ﻭ...)

۲۶- ﻧﺎﺭﺳﺎﻳﯽ ﻣﺰﻣﻦ ﮐﻠﻴﻪ ﺑﺎ ﻋﻼﺋﻢ ﻭﺍﺿﺢ ﺑﺎﻟﻴﻨﯽ ﻭ ﭘﺎﺭﺍﮐﻠﻴﻨﻴﮑﯽ ﺑﻪ ﺗﺄﻳﻴﺪ ﻣﺘﺨﺼﺺ ﻧﻔﺮﻭﻟﻮﮊﯼ.

۲۷- ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺍﻟﺘﻬﺎﺑﯽ ﻣﺰﻣﻦ ﺭﻭﺩﻩ (ﮐﺮﻭﻥ, ﮐﻮﻟﻴﺖ ﺍﻭﻟﺴﺮﻭ) ﮐﻪ ﭘﺎﺳﺦ ﺑﻪ ﺩﺭﻣﺎﻥ ﻃﺒﯽ ﻧﺪﺍﺩﻩ ﻭ ﺩﭼﺎﺭ ﻋﻮﺍﺭﺽ ﮔﻮﺍﺭﺷﯽ ﺷﺪﻩ ﺑﺎﺷﺪ.

۲۸- ﻫﻤﻮﻓﻴﻠﯽ ﻫﺎﯼ ﺷﺪﻳﺪ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﻣﺮﺍﻗﺒﺖ ﻫﺎﯼ ﻭﻳﮋﻩ ﺩﺍﺭﻧﺪ.

۲۹- ﮐﻮﻟﺴﺘﻮﻣﯽ ﻫﺎﯼ ﺩﺍﺋﻤﯽ.

۳۰- ﮐﻠﻴﻪ ﺍﻋﻤﺎﻝ ﺟﺮﺍﺣﯽ ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺩﺳﺘﮕﺎﻩ ﺍﺩﺭﺍﺭﯼ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﺗﻐﻴﻴﺮ ﻣﺴﻴﺮ ﺍﺩﺍﺭﯼ ﺑﺸﻮﺩ ﻭ ﺍﻧﻮﺍﻉ ﭘﻴﻮﻧﺪﻫﺎ ﻧﻈﻴﺮ ﭘﻴﻠﻮﺳﺘﻮﻣﯽ, ﻧﻔﺮﻭﺳﺘﻮﻣﯽ, ﺍﻭﺭﺗﺮﻭﺳﺘﻮﻣﯽ-ﺟﻠﺪﯼ, ﺳﻴﺴﺘﻮﺳﺘﻮﻣﯽ.

۳۱- ﻓﻴﺴﺘﻮﻝ ﻫﺎﯼ ﻣﺜﺎﻧﻪ ﺑﻪ ﺭﮐﺘﻮﻡ ﻏﻴﺮ ﻗﺎﺑﻞ ﺩﺭﻣﺎﻥ.

۳۲- ﺑﺮﺩﺍﺷﺘﻦ ﮐﺎﻣﻞ ﻣﺜﺎﻧﻪ.

۳۳- ﺑﺮﺩﺍﺷﺘﻦ ﻫﺮ ﺩﻭ ﮐﻠﻴﻪ ﻭ ﻳﺎ ﻳﮏ ﮐﻠﻴﻪ ﺑﻪ ﻫﻤﺮﺍﻩ ﺍﺧﺘﻼﻝ ﻋﻤﻠﮑﺮﺩ ﺩﺭ ﮐﻠﻴﻪ ﻣﻘﺎﺑﻞ.

۳۴- ﭘﻴﻮﻧﺪ ﮐﻠﻴﻪ.

۳۵- ﻓﻘﺪﺍﻥ ﻫﺮ ﺩﻭ ﭼﺸﻢ ﻭ ﻳﺎ ﺁﻥ ﮐﻪ ﻫﺮ ﺩﻭ ﭼﺸﻢ ﻋﻤﻼ ﻓﺎﻗﺪ ﺑﻴﻨﺎﻳﯽ ﺑﺎﺷﺪ.

ﺗﺒﺼﺮﻩ: ﺑﻪ ﻫﺮ ﻋﻠﺘﯽ ﮐﻪ ﺩﻳﺪ ﻳﮏ ﭼﺸﻢ ﺑﻪ ﻣﻴﺰﺍﻥ ۱۰۰% ﻭ ﺩﻳﺪ ﭼﺸﻢ ﺩﻳﮕﺮ ﺑﻪ ﻣﻴﺰﺍﻥ ۵۰% ﺍﺯ ﺑﻴﻦ ﺭﻓﺘﻪ ﺑﺎﺷﺪ ﻭ ﻳﺎ ﺩﻳﺪ ﻫﺮ ﺩﻭ ﭼﺸﻢ ﺩﺭ ﻣﺠﻤﻮﻉ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ۵۰% ﺩﻳﺪ ﻳﮏ ﭼﺸﻢ ﺑﺎﺷﺪ ﺍﻳﻦ ﻧﻮﻉ ﻣﺼﺪﺍﻕ ﺩﺍﺭﺩ.

۳۶- ﮐﺮﯼ ﮐﺎﻣﻞ ﻫﺮ ﺩﻭ ﮔﻮﺵ ﮐﻪ ﺑﺎ ﺳﻤﻌﮏ ﻧﻴﺰ ﻣﻴﺰﺍﻥ ﮐﺎﻫﺶ ﺷﻨﻮﺍﻳﯽ ﺍﺯ ۶۰ ﺩﺳﯽ ﺑﻞ ﻣﺠﻤﻮﻋﺎ ﺩﺭ ﻫﺮ ﺩﻭ ﮔﻮﺵ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ.

۳۷- ﮐﺮﯼ ﻭ ﻻﻟﯽ ﺗﻮﺃﻡ ﺑﺎ ﺗﺄﻳﻴﺪ ﻣﺘﺨﺼﺺ ﮔﻮﺵ, ﺣﻠﻖ ﻭ ﺑﻴﻨﯽ.

۳۸- ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﻣﺎﺩﺭﺯﺍﺩﯼ, ﺍﮐﺘﺴﺎﺑﯽ, ﺗﻮﻣﻮﺭﺍﻝ, ﻳﺎ ﺍﻟﺘﻬﺎﺑﯽ (ﺭﻭﻣﺎﺗﻴﺴﻤﺎﻝ) ﻗﻠﺒﯽ, ﺁﺋﻮﺭﺕ ﻭ ﻳﺎ ﺳﺮﺧﺮﮒ ﻫﺎﯼ ﺑﺰﺭﮒ ﺩﻳﮕﺮ ﻣﺎﻧﻨﺪ ﮐﻮﺍﺭﮐﺘﺎﺳﻴﻮﻥ, ﺁﻧﻮﺭﻳﺴﻢ ﻫﺎ, ﻭ ﻳﺎ ﺍﺗﺴﺎﻉ ﺳﺮﺧﺮﮒ ﻫﺎﯼ ﺭﻳﻮﯼ, ﮐﺎﻧﺎﻝ ﺁﺭﺗﺮﻳﻞ ﺑﺎﺯ, ﺗﻨﮕﯽ ﺷﺮﻳﺎﻥ ﺭﻳﻮﯼ ﺍﺻﻠﯽ, ﻓﻴﺴﺘﻮﻝ ﻫﺎﯼ ﺷﺮﻳﺎﻧﯽ ﻭ ﺭﻳﻮﯼ ﺩﺭ ﺻﻮﺭﺕ ﻋﺎﺭﺿﻪ ﺩﺍﺭ ﺑﻮﺩﻥ.

۳۹- ﺑﻴﻤﺎﺭﯼ ﻋﺮﻭﻕ ﮐﺮﻭﻧﺮ (ﺍﻧﻔﺎﺭﮐﺘﻮﺱ ﻣﻴﻮﮐﺎﺭﺩ ﻭ ﺁﻧﮋﻳﻦ ﺻﺪﺭﯼ) ﮐﻪ ﻣﻮﺟﺐ ﺩﺭﺟﺎﺗﯽ ﺍﺯ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺒﯽ ﻭ ﺍﺯ ﮐﺎﺭﺍﻓﺘﺎﺩﮔﯽ ﻓﺮﺩ ﮔﺮﺩﺩ ﺑﺎ ﺗﺸﺨﻴﺺ ﻣﺮﺍﮐﺰ ﺗﺨﺼﺼﯽ ﻧﻴﺮﻭﻫﺎﯼ ﻣﺴﻠﺢ.

۴۰- ﻧﺎﺭﺳﺎﻳﯽ ﭘﻴﺸﺮﻓﺘﻪ ﻗﻠﺐ ﺑﺎ ﻋﻼﺋﻢ ﺑﺎﻟﻴﻨﯽ ﻭﺍﺿﺢ.

۴۱- ﺍﻋﻤﺎﻝ ﭘﻴﻮﻧﺪ ﻋﺮﻭﻕ ﮐﺮﻭﻧﺮ ﻭ ﺗﻌﻮﻳﺾ ﺩﺭﻳﭽﻪ ﻫﺎﯼ ﻗﻠﺐ ﻭ ﮔﺬﺍﺷﺘﻦ ﭘﻴﻦ ﻣﻴﮑﺮﻫﺎﯼ ﺩﺍﺋﻤﯽ ﮐﻪ ﻋﻠﻴﺮﻏﻢ ﺍﻧﺠﺎﻡ ﻋﻤﻞ ﺩﺭﺟﺎﺗﯽ ﺍﺯ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺐ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

۴۲- ﺍﻧﻮﺍﻉ ﺁﺭﻳﺘﻤﯽ ﻫﺎ ﻭ ﺑﻠﻮﮎ ﻫﺎﯼ ﻗﻠﺒﯽ ﮐﻪ ﻣﻮﺟﺐ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺒﯽ ﺷﺪﻩ ﻭ ﺍﻣﮑﺎﻥ ﺍﻧﺠﺎﻡ ﻓﻌﺎﻟﻴﺖ ﺭﻭﺯﻣﺮﻩ ﺭﺍ ﺍﺯ ﺷﺨﺺ ﺑﮕﻴﺮﺩ.

۴۳- ﺍﻧﻮﺍﻉ ﮐﺎﺭﺩﻳﻮﻣﻴﻮﭘﺎﺗﯽ ﻫﺎ.

۴۴- ﺍﻓﺰﺍﻳﺶ ﺍﻭﻟﻴﻪ ﻳﺎ ﺛﺎﻧﻮﻳﻪ ﻓﺸﺎﺭ ﺧﻮﻥ ﺷﺮﻳﺎﻥ ﺭﻳﻮﯼ, ﺗﻮﺃﻡ ﺑﺎ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺒﯽ ﺑﻪ ﻫﺮ ﻋﻠﺖ ﮐﻪ ﺑﺎﺷﺪ (ﻣﻨﻈﻮﺭ ﺍﺯ ﻧﺎﺭﺳﺎﻳﯽ ﻗﻠﺒﯽ ﻗﺮﺍﺭﮔﺮﻓﺘﻦ ﺷﺨﺺ ﺩﺭ ﮐﻼﺱ ﻫﺎﯼ ۳ ﻭ ۴ ﺗﻨﮕﯽ ﻧﻔﺲ ﻣﯽ ﺑﺎﺷﺪ)
