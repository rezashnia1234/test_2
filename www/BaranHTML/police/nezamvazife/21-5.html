ﺁﺋﻴﻦ ﻧﺎﻣﻪ ﻣﻌﺎﻳﻨﻪ ﻭ ﻣﻌﺎﻓﻴﺖ ﭘﺰﺷﮑﯽ

ﻣﺎﺩﻩ ۳۳ - ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺭﻭﺍﻧﭙﺰﺷﮑﯽ

ﻣﺸﻤﻮﻻﻥ ﻣﺒﺘﻼ ﺑﻪ ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺭﻭﺍﻧﭙﺰﺷﮑﯽ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻧﻮﻉ ﺑﻴﻤﺎﺭﯼ ﮐﻪ ﺑﻪ ﺁﻥ ﻣﺒﺘﻼ ﻣﯽ ﺑﺎﺷﻨﺪ ﺍﺯ ﻣﻌﺎﻓﻴﺖ ﻫﺎﯼ ﻣﻨﺪﺭﺝ ﺩﺭ ﻫﺮ ﺑﻨﺪ ﺑﻪ ﺷﺮﺡ ﺯﻳﺮ ﺍﺳﺘﻔﺎﺩﻩ ﺧﻮﺍﻫﻨﺪ ﻧﻤﻮﺩ:

ﺑﻨﺪ ۱: ﺍﻧﻮﺍﻉ ﭘﺴﻴﮑﻮﺯﻫﺎ ﺷﺎﻣﻞ (ﺍﺳﮑﻴﺰﻭﺍﻓﮑﺘﻴﻮ, ﺍﺧﺘﻼﻻﺕ ﻫﺬﻳﺎﻧﯽ, ﭘﺴﻴﮑﻮﺯﻫﺎﯼ ﺁﺗﻴﭙﻴﮏ ﻣﺰﻣﻦ), ﻣﻌﺎﻑ ﺩﺍﺋﻢ.

ﺑﻨﺪ ۲: ﺍﺧﺘﻼﻻﺕ ﺧﻠﻘﯽ:

ﺍﻟﻒ) ﺷﺪﻳﺪ ﺩﺭ ﻓﺎﺯﻣﺎﻧﻴﮏ ﻳﺎ ﻓﺎﺯ ﺍﻓﺴﺮﺩﮔﯽ ﺍﻋﻢ ﺍﺯ ﺣﺎﺩ ﻭ ﻳﺎ ﻣﺰﻣﻦ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺏ)ﺍﺧﺘﻼﻝ ﺍﻓﺴﺮﺩﮔﯽ ﺍﺳﺎﺳﯽ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺝ) ﺍﻧﻮﺍﻉ ﺍﺧﺘﻼﻝ ﺧﻠﻔﯽ ﺍﻓﺴﺮﺩﮔﯽ ﺧﻔﻴﻒ (ﺩﻳﺲ ﺗﺎﻳﻤﯽ ﻭ ﺳﺎﻳﮑﻠﻮﺗﺎﻳﻤﯽ), ﺍﺧﺘﻼﻝ ﺧﻠﻘﯽ ﻧﻮﻉ ﺩﻭﻡ (ﻫﻴﭙﻮﻣﺎﻧﻴﺎ) ﺩﺭﺻﻮﺭﺗﯽ ﮐﻪ ﺑﺎ ﺩﺍﺭﻭ ﺗﺤﺖ ﮐﻨﺘﺮﻝ ﺑﻮﺩﻩ ﻭ ﻳﺎ ﺗﻬﺪﻳﺪﮐﻨﻨﺪﻩ ﺧﻮﺩ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻥ ﻧﺒﺎﺷﺪ, ﻣﻌﺎﻑ ﺍﺯ ﺭﺯﻡ.

ﺑﻨﺪ ۳: ﺍﻧﻮﺍﻉ ﺍﺧﺘﻼﻻﺕ ﺷﺨﺼﻴﺘﯽ:

ﺍﻟﻒ) ﺍﻧﻮﺍﻉ ﺳﺎﻳﮑﻮﭘﺎﺕ ﻭ ﻧﻮﻉ ﻣﺮﺯﯼ ﻭ ﻧﻮﻉ ﺍﺳﮑﻴﺰﻭﺋﻴﺪ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺏ) ﺍﻧﻮﺍﻉ ﺩﻳﮕﺮ ﺍﺧﺘﻼﻝ ﺷﺨﺼﻴﺖ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻧﺎﺗﻮﺍﻥ ﮐﻨﻨﺪﻩ ﺑﺎﺷﺪ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺝ) ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻧﺎﺗﻮﺍﻥ ﮐﻨﻨﺪﻩ ﻧﺒﺎﺷﺪ, ﻣﻌﺎﻑ ﺍﺯ ﺭﺯﻡ.

ﺑﻨﺪ ۴: ﺍﻧﻮﺍﻉ ﻧﻮﺭﻭﺯﻫﺎﯼ ﺷﺪﻳﺪ ﻭ ﻣﻘﺎﻭﻡ ﺑﻪ ﺩﺭﻣﺎﻥ ﻭ ﻣﺰﺍﺣﻢ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.

ﺑﻨﺪ ۵: ﺍﻧﻮﺍﻉ ﻧﻘﻴﺼﻪ ﻫﺎﯼ ﻋﻘﻼﻧﯽ ﻭ ﮐﻨﺪﺫﻫﻨﯽ ﻫﺎ (ﻋﻘﺐ ﻣﺎﻧﺪﮔﯽ):

ﺍﻟﻒ) ﺑﺎ ﺿﺮﻳﺐ ﻫﻮﺷﯽ ﺯﻳﺮ ۶۰, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺏ) ﺑﺎﺿﺮﻳﺐ ﻫﻮﺷﯽ ﺑﻴﻦ ۷۵-۶۰, ﻣﻌﺎﻑ ﺍﺯ ﺭﺯﻡ.

ﺑﻨﺪ ۶: ﺍﺧﺘﻼﻻﺕ ﺗﮑﻠﻢ ﻭ ﻟﮑﻨﺖ ﺯﺑﺎﻥ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻋﻠﺖ ﺭﻭﺍﻧﯽ ﭘﺎﻧﻮﺭﻭﻟﻮﮊﻳﮏ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ:

ﺍﻟﻒ) ﺩﺭﺻﻮﺭﺕ ﺷﺪﻳﺪ ﺑﻮﺩﻥ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺏ) ﺩﺭ ﻣﻮﺍﺭﺩ ﺧﻔﻴﻒ, ﺧﺪﻣﺎﺕ ﻏﻴﺮ ﺭﺯﻣﯽ.

ﺑﻨﺪ ۷: ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺭﻭﺍﻧﯽ ﻧﺎﺷﯽ ﺍﺯ ﺟﻨﮓ ﻭ ﺣﻮﺍﺩﺙ:

ﺍﻟﻒ) ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﺳﺘﻘﺮﺍﺭ ﻳﺎﻓﺘﻪ ﺑﺎﺷﺪ ﺑﻪ ﺷﺮﻁ ﺛﺒﻮﺕ ﺩﺭ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻧﻈﺎﻣﯽ ﻳﺎ ﺩﺍﻧﺸﮕﺎﻫﯽ ﺗﻮﺳﻂ ﺩﻭ ﻧﻔﺮ ﺭﻭﺍﻧﭙﺰﺷﮏ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.
ﺏ) ﺩﺭ ﺻﻮﺭﺕ ﻋﺪﻡ ﺍﺳﺘﻘﺮﺍﺭ, ﺧﺪﻣﺎﺕ ﻏﻴﺮﺭﺯﻣﯽ.

ﺑﻨﺪ ۸: ﺍﺧﺘﻼﻝ ﺭﻓﺘﺎﺭ (ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﻋﺼﺒﯽ ﻭ ﺭﻭﺍﻧﯽ) ﻭ ﮐﮋﺧﻮﻳﯽ ﻫﺎ ﺑﻪ ﻃﻮﺭﯼ ﮐﻪ ﻣﻐﺎﻳﺮ ﺷﺌﻮﻧﺎﺕ ﻧﻈﺎﻣﯽ ﺑﺎﺷﺪ ﻫﻤﭽﻨﻴﻦ ﺍﻧﺤﺮﺍﻓﺎﺕ ﺍﺧﻼﻗﯽ ﻭ ﺟﻨﺴﯽ ﻣﺎﻧﻨﺪ ﺗﺮﺍﻧﺲ ﺳﮑﺴﻮﺍﻟﻴﺴﻢ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.

ﺑﻨﺪ ۹: ﺍﺧﺘﻼﻻﺕ ﮐﻨﺘﺮﻝ ﺗﮑﺎﻧﻪ, ﻣﻌﺎﻑ ﺩﺍﺋﻢ.

ﺗﺒﺼﺮﻩ: ﺗﻴﮏ ﻫﺎ (ﺗﮑﺎﻧﻪ ﻫﺎﯼ) ﺿﻌﻴﻒ ﻭ ﮔﺬﺭﺍ ﻣﺜﻞ ﭘﻠﮏ ﺯﺩﻥ ﻭ..., ﻣﻌﺎﻑ ﺍﺯﺭﺯﻡ.
